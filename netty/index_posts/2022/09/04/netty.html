Blocking & UnBlocking

Andy Sun 4 Sep 2022

Part I Blocking mode

Server

// single thread
// 0. ByteBuffer
ByteBuffer buffer = ByteBuffer.allocate(16);
// 1. create server
ServerSocketChannel ssc = ServerSocketChannel.open();

// 2.bind port
ssc.bind(new InetSocketAddress(8080));

// 3. init collection of connection
List<SocketChannel> channels = new ArrayList<>();
while (true) {
    // 4. accept is used for creating connection with client， SocketChannel is used for communication
    log.debug("connecting...");
    SocketChannel sc = ssc.accept(); // block，stop current thread
    log.debug("connected... {}", sc);
    channels.add(sc);
    for (SocketChannel channel : channels) {
        // 5. receive data from client
        log.debug("before read... {}", channel);
        channel.read(buffer); // block，stop current thread
        buffer.flip();// switch write mode to read mode
        debugRead(buffer);
        buffer.clear();// reset
        log.debug("after read...{}", channel);
    }
}

Client

SocketChannel sc = SocketChannel.open();
sc.connect(new InetSocketAddress("localhost", 8080));
System.out.println("waiting...");

ssc.accept() and channel.read(buffer) will both block current thread .

Part II , unblocking mode

// 0. ByteBuffer
ByteBuffer buffer = ByteBuffer.allocate(16);
// 1. create server
ServerSocketChannel ssc = ServerSocketChannel.open();
ssc.configureBlocking(false); // no blocking
// 2. bind port
ssc.bind(new InetSocketAddress(8080));
// 3. connection collection
List<SocketChannel> channels = new ArrayList<>();
while (true) {
    // 4.  accept is used for creating connection with client， SocketChannel is used for communication
    SocketChannel sc = ssc.accept(); // no blocking，current thread will go on running，but sc would be null if no connection now
    if (sc != null) {
        log.debug("connected... {}", sc);
        sc.configureBlocking(false); // no blocking mode, will cause channel.read(buffer) not block current thread.
        channels.add(sc);
    }
    for (SocketChannel channel : channels) {
        // 5. receive data from client
        int read = channel.read(buffer);// no blocking，current thread will go on running，but read will return 0 if no data now
        if (read > 0) {
            buffer.flip();
            debugRead(buffer);
            buffer.clear();
            log.debug("after read...{}", channel);
        }
    }
}
